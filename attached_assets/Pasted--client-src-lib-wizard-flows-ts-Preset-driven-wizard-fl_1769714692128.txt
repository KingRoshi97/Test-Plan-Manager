// client/src/lib/wizard/flows.ts
// Preset-driven wizard flows (Category → Mode → Preset).
// These flows are designed for BOTH beginners and devs. Beginner help/examples included.
// You can render these dynamically to build the wizard UI.

export type Category = "web" | "mobile" | "api" | "library" | "automation" | "game";
export type Mode =
  | "new_build"
  | "existing_upgrade"
  | "ui_overhaul"
  | "refactor_hardening"
  | "add_feature_module";

export type FieldUI =
  | "text"
  | "textarea"
  | "select"
  | "toggle"
  | "chips"
  | "list"
  | "file_docs"
  | "file_zip";

export type DependsOn = { key: string; equals: any };

export type FieldSpec = {
  key: string; // draft path key (string path, ex: "project.name" or "features.p0")
  label: string;
  ui: FieldUI;
  required?: boolean;
  placeholder?: string;
  help?: string;
  examples?: string[];
  options?: string[];
  dependsOn?: DependsOn;
};

export type FieldGroup = {
  id: string;
  title?: string;
  description?: string;
  fields: FieldSpec[];
};

export type WizardStep = {
  id: string;
  title: string;
  description?: string;
  required?: boolean; // step gating
  fieldGroups: FieldGroup[];
};

export type WizardFlow = {
  id: string;
  label: string;
  category: Category;
  steps: WizardStep[];
};

export const flowRegistry: Record<Category, WizardFlow> = {
  web: flowWeb,
  mobile: flowMobile,
  api: flowApi,
  library: flowLibrary,
  automation: flowAutomation,
  game: flowGame
};

// ------------------------------
// Mode application (overlays steps + required fields)
// ------------------------------

export function applyModeToFlow(base: WizardFlow, mode: Mode): WizardFlow {
  // Deep clone to avoid mutating base flows
  const flow: WizardFlow = JSON.parse(JSON.stringify(base));

  // Helpers
  const findStep = (id: string) => flow.steps.find((s) => s.id === id);
  const requireField = (key: string) => {
    for (const step of flow.steps) {
      for (const group of step.fieldGroups) {
        for (const field of group.fields) {
          if (field.key === key) field.required = true;
        }
      }
    }
  };

  const upsertStepAfter = (afterStepId: string, step: WizardStep) => {
    const idx = flow.steps.findIndex((s) => s.id === afterStepId);
    const existingIdx = flow.steps.findIndex((s) => s.id === step.id);
    if (existingIdx !== -1) flow.steps.splice(existingIdx, 1);
    flow.steps.splice(idx + 1, 0, step);
  };

  // Mode changes
  if (mode === "existing_upgrade") {
    // Require ZIP upload and add regression + validation step
    requireField("projectPackage.id");

    const zipStep: WizardStep = {
      id: "zip_required",
      title: "Upload project ZIP",
      description:
        "Required for upgrades. We scan it (framework/scripts/deps) so the agent doesn’t guess.",
      required: true,
      fieldGroups: [
        {
          id: "zip",
          title: "Project ZIP",
          fields: [
            { ...F.zipUpload(), required: true },
            {
              key: "projectPackage.notes",
              label: "Notes about the current project (optional)",
              ui: "textarea",
              required: false,
              help: "Anything special about the codebase? (monorepo, legacy areas, known issues)",
              examples: ["Monorepo with /client and /server", "Navigation contracts are frozen", "Old auth code is fragile"]
            }
          ]
        }
      ]
    };

    const regressionStep: WizardStep = {
      id: "regression",
      title: "Regression & validation",
      description: "What must stay working after changes? How should we verify?",
      required: true,
      fieldGroups: [
        {
          id: "regression",
          fields: [
            { ...F.regressionMustNotBreak(), required: true },
            F.validationCommands()
          ]
        }
      ]
    };

    // Insert zip step after basics
    upsertStepAfter("basics", zipStep);

    // Insert regression step after intent (or after zip if intent absent)
    const after = findStep("intent") ? "intent" : "zip_required";
    upsertStepAfter(after, regressionStep);

    // Adjust review description
    const review = findStep("review");
    if (review) {
      review.description =
        "Review the upgrade request. The kit will include migration notes + regression checklist.";
    }
  }

  if (mode === "ui_overhaul") {
    // Insert design-focused step after UX
    const uiStep: WizardStep = {
      id: "ui_overhaul",
      title: "UI overhaul details",
      description: "Define the design direction so the output matches your brand and quality bar.",
      required: true,
      fieldGroups: [
        {
          id: "ui",
          fields: [
            {
              key: "ui.goal",
              label: "What should the UI feel like?",
              ui: "textarea",
              required: true,
              placeholder: "Clean, premium, professional…",
              examples: ["Dark-first, glass panels, strong typography, minimal clutter."]
            },
            {
              key: "ui.components",
              label: "Key components to standardize",
              ui: "list",
              required: false,
              examples: ["Buttons", "Cards", "Tables", "Modals", "Forms", "Toasts"]
            },
            {
              key: "ui.mustKeep",
              label: "What must stay the same?",
              ui: "list",
              required: false,
              examples: ["Navigation structure", "Route names", "Brand colors"]
            }
          ]
        }
      ]
    };

    // Try to place after ux step if present, else after scope
    const anchor = findStep("ux") ? "ux" : "scope";
    upsertStepAfter(anchor, uiStep);

    const review = findStep("review");
    if (review) review.description = "Review UI overhaul goals. The kit will include UI changelog + component adoption plan.";
  }

  if (mode === "refactor_hardening") {
    // Insert hardening step after intent
    const hardeningStep: WizardStep = {
      id: "hardening",
      title: "Hardening targets",
      description: "Performance, security, reliability—what should improve?",
      required: true,
      fieldGroups: [
        {
          id: "hardening",
          fields: [
            { ...F.hardeningTargets(), required: true },
            { ...F.testExpectations(), required: true },
            {
              key: "hardening.rollout",
              label: "Rollout strategy (optional)",
              ui: "select",
              required: false,
              options: ["Not sure", "Ship all at once", "Feature flags", "Phased rollout"],
              help: "If unsure, choose Feature flags."
            }
          ]
        }
      ]
    };

    const after = findStep("intent") ? "intent" : "basics";
    upsertStepAfter(after, hardeningStep);

    const review = findStep("review");
    if (review) review.description = "Review hardening goals. The kit will include a hardening plan + test strategy.";
  }

  if (mode === "add_feature_module") {
    // Insert module scope step after scope/features if present
    const moduleStep: WizardStep = {
      id: "feature_module",
      title: "Feature module scope",
      description: "Define the new subsystem so it integrates cleanly.",
      required: true,
      fieldGroups: [
        {
          id: "module",
          fields: [
            {
              key: "module.name",
              label: "Module name",
              ui: "text",
              required: true,
              placeholder: "e.g., Billing, Chat, Events"
            },
            {
              key: "module.description",
              label: "Module description",
              ui: "textarea",
              required: true,
              placeholder: "What is this module and what does it do?"
            },
            {
              key: "module.integrations",
              label: "Integration points",
              ui: "list",
              required: false,
              help: "What existing parts does it touch?",
              examples: ["User accounts", "Payments", "Notifications", "Admin dashboard"]
            },
            {
              key: "module.data",
              label: "New data entities (optional)",
              ui: "list",
              required: false,
              examples: ["Subscription", "Invoice", "Message", "Event"]
            }
          ]
        }
      ]
    };

    const anchor = findStep("scope") ? "scope" : "intent";
    upsertStepAfter(anchor, moduleStep);

    const review = findStep("review");
    if (review) review.description = "Review the new module scope. The kit will focus on integration and acceptance criteria.";
  }

  return flow;
}

// ------------------------------
// Utility: get flow by category and apply mode
// ------------------------------

export function getFlow(category: Category, mode: Mode): WizardFlow {
  const base = flowRegistry[category];
  return applyModeToFlow(base, mode);
}
