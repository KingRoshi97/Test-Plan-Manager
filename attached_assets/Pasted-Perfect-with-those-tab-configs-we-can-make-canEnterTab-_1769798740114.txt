2) Implement canEnterTab() using “prior steps must be valid”

This is the key change.

const canEnterTab = useCallback(
  (tabId: string) => {
    // Always allow "type"
    if (tabId === "type") return true;

    // If no flow yet, block everything else
    if (!wizardFlow) return false;

    const targetPos = getTabPos(tabId);
    if (targetPos < 0) return false;

    // Validate each PRIOR tab's step in order
    for (let i = 0; i < targetPos; i++) {
      const priorTab = currentTabs[i];
      if (priorTab.id === "type") continue;

      const priorStepId = priorTab.stepId || priorTab.id;
      const priorStep = wizardFlow.steps.find(s => s.id === priorStepId);
      if (!priorStep) continue; // if mismatch, don't hard-block

      // ---- ZIP gating (your modes add upload tab; use it when present)
      // Treat any step containing a required file_zip field as "must have ready package"
      const priorHasRequiredZip = priorStep.fieldGroups
        .flatMap(g => g.fields)
        .some(f => f.ui === "file_zip" && f.required);

      if (priorHasRequiredZip && !isPackageReady) return false;

      // ---- Step validation (field requiredness)
      // You need ONE shared validator. If you already have validateStep(draft, step), use it here.
      // If not, implement minimal required-field validation below.
      const ok = validateStepMinimal(wizardDraft, priorStep);
      if (!ok) return false;
    }

    return true;
  },
  [wizardFlow, currentTabs, wizardDraft, isPackageReady, getTabPos]
);

Minimal validator (drop-in)

This keeps you unblocked even if you don’t have a reusable validator exported from flows.ts.

function getByPath(obj: any, path: string) {
  return path.split(".").reduce((acc, k) => (acc == null ? undefined : acc[k]), obj);
}

function isEmptyValue(v: any) {
  if (v == null) return true;
  if (typeof v === "string") return v.trim().length === 0;
  if (Array.isArray(v)) return v.length === 0;
  return false;
}

function validateStepMinimal(draft: any, step: any) {
  for (const group of step.fieldGroups || []) {
    for (const field of group.fields || []) {
      if (!field.required) continue;

      // dependsOn support (simple)
      if (field.dependsOn) {
        const depVal = getByPath(draft, field.dependsOn.key);
        const expected = field.dependsOn.value;
        if (depVal !== expected) continue; // not applicable
      }

      // For file_docs: require at least one id
      if (field.ui === "file_docs") {
        const v = getByPath(draft, field.key);
        if (!Array.isArray(v) || v.length === 0) return false;
        continue;
      }

      // For other fields: basic non-empty
      const v = getByPath(draft, field.key);
      if (isEmptyValue(v)) return false;
    }
  }
  return true;
}

3) Update goNext/goPrev to skip disabled tabs (using the new canEnterTab)
const goNext = useCallback(() => {
  const currentIdx = currentTabs.findIndex(t => t.id === currentStep);
  if (currentIdx < 0) return;

  for (let i = currentIdx + 1; i < currentTabs.length; i++) {
    const nextId = currentTabs[i].id;
    if (canEnterTab(nextId)) {
      setCurrentStep(nextId);
      return;
    }
  }
}, [currentTabs, currentStep, canEnterTab]);

const goPrev = useCallback(() => {
  const currentIdx = currentTabs.findIndex(t => t.id === currentStep);
  if (currentIdx < 0) return;

  for (let i = currentIdx - 1; i >= 0; i--) {
    const prevId = currentTabs[i].id;
    if (canEnterTab(prevId)) {
      setCurrentStep(prevId);
      return;
    }
  }
}, [currentTabs, currentStep, canEnterTab]);
