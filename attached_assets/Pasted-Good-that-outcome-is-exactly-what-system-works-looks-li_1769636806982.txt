Implementation (Node) — add one script + 1 dependency
1) Install dependency

Use archiver (common, reliable):

npm i archiver

2) Add npm script

In package.json:

{
  "scripts": {
    "roshi:package": "node scripts/roshi-package.mjs"
  }
}

3) Create scripts/roshi-package.mjs

This follows your established patterns: parse args, validate domain, build file list, create dist, zip, ROSHI_REPORT.

// scripts/roshi-package.mjs
import fs from "node:fs";
import path from "node:path";
import crypto from "node:crypto";
import archiver from "archiver";

const ROOT = process.cwd();

function parseArgs(argv) {
  const args = { domain: null, dryRun: false, force: false };
  for (let i = 2; i < argv.length; i++) {
    const a = argv[i];
    if (a === "--dry-run") args.dryRun = true;
    else if (a === "--force") args.force = true;
    else if (a === "--domain") args.domain = argv[++i] ?? null;
  }
  return args;
}

function readJson(p) {
  return JSON.parse(fs.readFileSync(p, "utf8"));
}

function exists(p) {
  try { fs.accessSync(p); return true; } catch { return false; }
}

function ensureDir(p) {
  fs.mkdirSync(p, { recursive: true });
}

function sha256(buf) {
  return crypto.createHash("sha256").update(buf).digest("hex");
}

function walk(dir) {
  const out = [];
  const entries = fs.readdirSync(dir, { withFileTypes: true });
  for (const e of entries) {
    const full = path.join(dir, e.name);
    if (e.isDirectory()) out.push(...walk(full));
    else out.push(full);
  }
  return out;
}

function rel(p) {
  return path.relative(ROOT, p).replaceAll("\\", "/");
}

function isUnder(p, parent) {
  const rp = path.resolve(p);
  const rpar = path.resolve(parent);
  return rp === rpar || rp.startsWith(rpar + path.sep);
}

function collectFiles({ domain }) {
  const files = [];

  // Always include core workspace roots
  const mustRoots = [
    "workspace/roshi/domains.json",
    "workspace/roshi/sources.json",
    "workspace/docs/roshi_v2/README.md",
    "workspace/docs/roshi_v2/00_product",
    "workspace/docs/roshi_v2/00_registry"
  ];

  // Domain packs
  if (domain) {
    mustRoots.push(`workspace/docs/roshi_v2/02_domains/${domain}`);
  } else {
    mustRoots.push("workspace/docs/roshi_v2/02_domains");
  }

  // Optional: include inputs for provenance (safe, useful)
  const optionalRoots = [
    "workspace/docs/inputs/TARGET_OUTLINE.md"
    // Add the PDF copy here if you decide to include it:
    // "workspace/docs/inputs/Roshi system start up guide  (1).pdf"
  ];

  const addRoot = (r, required) => {
    const abs = path.join(ROOT, r);
    if (!exists(abs)) {
      if (required) throw new Error(`Missing required path: ${r}`);
      return;
    }
    const stat = fs.statSync(abs);
    if (stat.isDirectory()) {
      for (const f of walk(abs)) files.push(f);
    } else {
      files.push(abs);
    }
  };

  for (const r of mustRoots) addRoot(r, true);
  for (const r of optionalRoots) addRoot(r, false);

  // De-dupe
  return [...new Set(files)].sort();
}

function scanOpenQuestions(domain) {
  const base = domain
    ? path.join(ROOT, "workspace/docs/roshi_v2/02_domains", domain)
    : path.join(ROOT, "workspace/docs/roshi_v2/02_domains");

  if (!exists(base)) return { count: 0, paths: [] };

  const files = walk(base).filter(f => f.endsWith("OPEN_QUESTIONS.md"));
  return {
    count: files.length,
    paths: files.map(rel)
  };
}

async function zipFiles({ files, zipPath, rootPrefix = "roshi_bundle" }) {
  ensureDir(path.dirname(zipPath));

  const output = fs.createWriteStream(zipPath);
  const archive = archiver("zip", { zlib: { level: 9 } });

  const done = new Promise((resolve, reject) => {
    output.on("close", resolve);
    output.on("error", reject);
    archive.on("warning", (err) => {
      if (err.code === "ENOENT") return;
      throw err;
    });
    archive.on("error", reject);
  });

  archive.pipe(output);

  for (const abs of files) {
    // Enforce no zip-slip: only include files under workspace/
    const workspaceRoot = path.join(ROOT, "workspace");
    if (!isUnder(abs, workspaceRoot)) continue;

    const inside = path.posix.join(rootPrefix, rel(abs));
    archive.file(abs, { name: inside });
  }

  await archive.finalize();
  await done;
}

(async function main() {
  const args = parseArgs(process.argv);

  const report = {
    command: "roshi:package",
    ok: false,
    failed: 0,
    failures: [],
    created: [],
    modified: [],
    skipped: [],
    dryRun: args.dryRun,
    domain: args.domain,
    outputs: {}
  };

  try {
    const domainsPath = path.join(ROOT, "workspace/roshi/domains.json");
    if (!exists(domainsPath)) throw new Error("Missing workspace/roshi/domains.json");

    const domains = readJson(domainsPath);
    const domainSlugs = Array.isArray(domains.domains)
      ? domains.domains.map(d => d.slug)
      : [];

    if (args.domain && !domainSlugs.includes(args.domain)) {
      throw new Error(`Invalid domain '${args.domain}'. Must be one of: ${domainSlugs.join(", ")}`);
    }

    const distDir = path.join(ROOT, "dist");
    const bundleDir = path.join(distDir, "roshi_bundle");
    const zipOut = path.join(distDir, "roshi_bundle.zip");

    if (exists(zipOut) && !args.force) {
      report.skipped.push(rel(zipOut));
      report.ok = true;
      console.log(JSON.stringify(report, null, 2));
      return;
    }

    const openQ = scanOpenQuestions(args.domain);

    const manifest = {
      bundleVersion: "0.1.0",
      createdAt: new Date().toISOString(),
      project: {
        name: "Roshi Studio",
        intent: "api+web powered by @roshi/core"
      },
      pipeline: {
        state: "draft",
        domains: args.domain ? [args.domain] : domainSlugs
      },
      entryDocs: [
        "workspace/docs/roshi_v2/README.md",
        "workspace/docs/roshi_v2/00_product/PROJECT_OVERVIEW.md",
        "workspace/docs/roshi_v2/00_registry/domain-map.md"
      ],
      commandsToRun: [
        "npm run roshi:review",
        "npm run roshi:verify",
        "npm run roshi:lock"
      ],
      doNotTouch: [
        "workspace/docs/roshi_v2/00_registry/**"
      ],
      openQuestions: openQ,
      artifacts: {
        zipPath: "dist/roshi_bundle.zip",
        workspaceRoot: "workspace/"
      }
    };

    const agentPrompt = `# Roshi Agent Handoff

You are receiving a Roshi bundle generated by roshi:package.

## What this bundle contains
- Roshi v2 workspace docs under /workspace/docs/roshi_v2
- Registry/product docs and domain packs
- manifest.json is the source of truth

## Non-negotiable rules
- No invention: missing info stays UNKNOWN + logged to Open Questions
- Verify before lock
- No overwrite unless explicitly allowed
- Always output ROSHI_REPORT at the end of each run

## What to do next
1) Read:
   - workspace/docs/roshi_v2/README.md
   - workspace/docs/roshi_v2/00_product/PROJECT_OVERVIEW.md
   - workspace/docs/roshi_v2/00_registry/domain-map.md
2) Run:
   - npm run roshi:review
   - npm run roshi:verify
   - npm run roshi:lock

## Current Open Questions
Count: ${openQ.count}
Paths:
${openQ.paths.map(p => `- ${p}`).join("\n")}
`;

    const files = collectFiles({ domain: args.domain });

    if (args.dryRun) {
      report.ok = true;
      report.outputs = {
        wouldWrite: [
          "dist/roshi_bundle/manifest.json",
          "dist/roshi_bundle/agent_prompt.md",
          "dist/roshi_bundle.zip"
        ],
        includeCount: files.length
      };
      console.log(JSON.stringify(report, null, 2));
      return;
    }

    ensureDir(bundleDir);

    const manifestPath = path.join(bundleDir, "manifest.json");
    const promptPath = path.join(bundleDir, "agent_prompt.md");

    fs.writeFileSync(manifestPath, JSON.stringify(manifest, null, 2));
    fs.writeFileSync(promptPath, agentPrompt);

    report.created.push(rel(manifestPath), rel(promptPath));

    // Zip includes everything under workspace plus the bundle files (manifest/prompt)
    // We'll include bundle files by writing them into workspace-safe path OR add separately:
    // easiest: add them as non-workspace entries under roshi_bundle/ at zip root.
    // Here: zipFiles only includes workspace/*, so we also add manifest/prompt by copying into workspace/tmp.
    // To keep it simple: we’ll zip both workspace files and bundle files using archiver directly here:

    const output = fs.createWriteStream(zipOut);
    const archive = archiver("zip", { zlib: { level: 9 } });

    const done = new Promise((resolve, reject) => {
      output.on("close", resolve);
      output.on("error", reject);
      archive.on("warning", (err) => { if (err.code !== "ENOENT") reject(err); });
      archive.on("error", reject);
    });

    archive.pipe(output);

    // Add bundle top files
    archive.file(manifestPath, { name: "roshi_bundle/manifest.json" });
    archive.file(promptPath, { name: "roshi_bundle/agent_prompt.md" });

    // Add workspace files
    for (const abs of files) {
      const inside = path.posix.join("roshi_bundle", rel(abs));
      archive.file(abs, { name: inside });
    }

    await archive.finalize();
    await done;

    report.created.push(rel(zipOut));
    report.outputs.zipSha256 = sha256(fs.readFileSync(zipOut));
    report.ok = true;

    console.log(JSON.stringify(report, null, 2));
  } catch (e) {
    report.failed = 1;
    report.failures.push(String(e?.message || e));
    console.log(JSON.stringify(report, null, 2));
    process.exit(1);
  }
})();
